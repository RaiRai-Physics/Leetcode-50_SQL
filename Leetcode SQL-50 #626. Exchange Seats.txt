Leetcode SQL-50: #626. Exchange Seats

Question Link: https://leetcode.com/problems/exchange-seats/?envType=study-plan-v2&envId=top-sql-50
# Write your MySQL query statement below
SELECT
CASE
WHEN id % 2 = 1 AND id < (SELECT MAX(id) FROM Seat) THEN id + 1
WHEN id % 2 = 0 THEN id - 1
ELSE id
END AS id,
student
FROM Seat
ORDER BY id;
