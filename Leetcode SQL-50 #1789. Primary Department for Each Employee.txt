Leetcode SQL-50: #1789. Primary Department for Each Employee
Question Link: https://leetcode.com/problems/primary-department-for-each-employee/?envType=study-plan-v2&envId=top-sql-50

/* Write your T-SQL query statement below */
select
employee_id,
department_id
from Employee
where primary_flag = 'Y'
or employee_id in (
        select employee_id
        from Employee
        group by employee_id
        having count(*) = 1
   );
