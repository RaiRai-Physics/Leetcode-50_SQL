Leetcode SQL 50; #577. Employee Bonus
Question link: https://leetcode.com/problems/employee-bonus/submissions/1894979451/?envType=study-plan-v2&envId=top-sql-50
/* Write your T-SQL query statement below */
SELECT e.name, b.bonus
FROM Employee e
LEFT JOIN Bonus b
ON e.empId = b.empId
WHERE b.bonus < 1000
OR b.bonus IS NULL;
